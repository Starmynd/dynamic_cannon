const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); const roundText = document.getElementById('round'); const timeText = document.getElementById('time'); const shotsText = document.getElementById('shots'); const scoreText = document.getElementById('score'); const shootButton = document.getElementById('shootButton'); const roundMessage = document.getElementById('roundMessage'); let clouds = []; let cannon, wagon, shell = null; let shotsLeft = 20; let score = 0; let timer = 90; let round = 1; let beams = []; let timerInterval; const maxScore = 10; // Максимальное количество очков для завершения уровня // Создаем облака function createClouds() { clouds = []; for (let i = 0; i < 5; i++) { let cloud = { x: Math.random() * canvas.width, y: Math.random() * 100, size: Math.random() * 40 + 30, speed: Math.random() * 0.5 + 0.2 }; clouds.push(cloud); } } function drawClouds() { ctx.fillStyle = "rgba(255, 255, 255, 0.5)"; clouds.forEach(cloud => { ctx.beginPath(); ctx.arc(cloud.x, cloud.y, cloud.size, 0, Math.PI * 2); ctx.fill(); }); } function updateClouds() { clouds.forEach(cloud => { cloud.x -= cloud.speed; if (cloud.x < -cloud.size) { cloud.x = canvas.width + cloud.size; cloud.y = Math.random() * 100; cloud.size = Math.random() * 40 + 30; cloud.speed = Math.random() * 0.5 + 0.2; } }); } // Рисуем горы function drawMountains() { ctx.fillStyle = 'rgba(139, 69, 19, 1)'; ctx.beginPath(); ctx.moveTo(50, canvas.height); ctx.lineTo(canvas.width * 0.25, canvas.height * 0.4); ctx.lineTo(canvas.width * 0.5, canvas.height); ctx.closePath(); ctx.fill(); ctx.beginPath(); ctx.moveTo(canvas.width * 0.75, canvas.height); ctx.lineTo(canvas.width, canvas.height * 0.5); ctx.lineTo(canvas.width, canvas.height); ctx.closePath(); ctx.fill(); // Белые "шапки" на горах ctx.fillStyle = 'rgba(255, 255, 255, 0.7)'; ctx.beginPath(); ctx.moveTo(canvas.width * 0.25, canvas.height * 0.4); ctx.lineTo(canvas.width * 0.22, canvas.height * 0.5); ctx.lineTo(canvas.width * 0.28, canvas.height * 0.5); ctx.closePath(); ctx.fill(); ctx.beginPath(); ctx.moveTo(canvas.width * 0.9, canvas.height * 0.5); ctx.lineTo(canvas.width * 0.88, canvas.height * 0.55); ctx.lineTo(canvas.width * 0.92, canvas.height * 0.55); ctx.closePath(); ctx.fill(); } class Cannon { constructor() { this.width = canvas.width * 0.08; // Пропорция ширины this.height = canvas.height * 0.05; // Пропорция высоты this.x = canvas.width / 2 - this.width / 2; this.y = canvas.height - this.height - 5; this.speed = canvas.width * 0.15; // Скорость пропорциональна ширине this.direction = 1; } update(delta) { this.x += this.direction * this.speed * delta; if (this.x > canvas.width - this.width || this.x < 0) { this.direction *= -1; } } draw() { ctx.fillStyle = "#1E3A8A"; ctx.fillRect(this.x, this.y, this.width, this.height); ctx.fillStyle = "#3949AB"; ctx.fillRect(this.x + this.width / 4, this.y - this.height * 0.4, this.width / 2, this.height * 0.4); ctx.fillStyle = "#000000"; ctx.fillRect(this.x + this.width / 2 - 5, this.y - this.height, 10, this.height * 0.75); } } class Wagon { constructor() { this.width = canvas.width * 0.15; // Пропорция ширины this.height = canvas.height * 0.05; // Пропорция высоты this.x = canvas.width / 2 - this.width / 2; this.y = canvas.height * 0.55; // Веревка чуть выше середины экрана this.speed = canvas.width * 0.1; // Скорость пропорциональна ширине this.direction = -1; } update(delta) { this.x += this.direction * this.speed * delta; if (this.x > canvas.width - this.width || this.x < 0) { this.direction *= -1; } } draw() { ctx.strokeStyle = '#8B4513'; // Трос ctx.lineWidth = 2; ctx.beginPath(); ctx.moveTo(0, this.y - 10); // Линия троса ctx.lineTo(canvas.width, this.y - 10); ctx.stroke(); ctx.strokeStyle = '#8B4513'; // Крепление вагонетки ctx.beginPath(); ctx.moveTo(this.x + this.width / 2, this.y - 10); ctx.lineTo(this.x + this.width / 2, this.y); ctx.stroke(); ctx.fillStyle = '#FF69B4'; // Вагонетка ctx.fillRect(this.x, this.y, this.width, this.height); } } class Shell { constructor(x, y, vy) { this.radius = canvas.width * 0.015; // Пропорция радиуса this.x = x; this.y = y; this.vy = vy; this.active = true; this.bounced = false; // Отскочил ли мяч } update(delta) { this.y += this.vy * delta; // Отражение от балки beams.forEach(beam => { if (this.y - this.radius <= beam.y + beam.height && this.y - this.radius >= beam.y && this.x >= beam.x && this.x <= beam.x + beam.width) { this.vy = Math.abs(this.vy); // Убедимся, что мяч продолжает двигаться вверх this.bounced = true; } }); // Проверка на исчезновение мяча if (this.y < 0 || this.y > canvas.height) { this.active = false; } // Попадание по вагонетке if ( this.y + this.radius >= wagon.y && this.x >= wagon.x && this.x <= wagon.x + wagon.width ) { if (this.bounced) { score += 100; this.active = false; updateUI(); } else { this.vy = -Math.abs(this.vy); // Отскок от дна корзины, чтобы мяч падал вниз } } } draw() { ctx.fillStyle = "#000000"; ctx.beginPath(); ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2); ctx.fill(); } } class Beam { constructor(x, y, width, height) { this.x = x; this.y = y; this.width = width; this.height = height; } draw() { ctx.fillStyle = '#8B4513'; ctx.fillRect(this.x, this.y, this.width, this.height); } } function shoot() { if (shotsLeft > 0 && !shell) { const shellX = cannon.x + cannon.width / 2; const shellY = cannon.y; const velocityY = -800; shell = new Shell(shellX, shellY, velocityY); shotsLeft--; updateUI(); } } function updateUI() { shotsText.textContent = `Shots: ${shotsLeft}`; scoreText.textContent = `Score: ${score}`; timeText.textContent = `Time: ${Math.ceil(timer)}`; roundText.textContent = `Round: ${round}`; } function startTimer() { timerInterval = setInterval(() => { if (timer > 0) { timer--; updateUI(); } else { clearInterval(timerInterval); endGame("Time's up! Try again!", true); } }, 1000); } function endGame(message, restart) { clearInterval(timerInterval); roundMessage.textContent = message; roundMessage.style.display = 'block'; if (restart) { setTimeout(() => { resetGame(); }, 3000); } } function resetGame() { round = 1; score = 0; shotsLeft = 20; timer = 90; roundMessage.style.display = 'none'; startTimer(); } function init() { resizeCanvas(); createClouds(); cannon = new Cannon(); wagon = new Wagon(); beams = [ new Beam(50, 20, 200, 10), new Beam(canvas.width - 250, 20, 200, 10) ]; startTimer(); gameLoop(); } function gameLoop() { const delta = 1 / 60; ctx.clearRect(0, 0, canvas.width, canvas.height); updateClouds(); drawClouds(); drawMountains(); cannon.update(delta); wagon.update(delta); cannon.draw(); wagon.draw(); beams.forEach(beam => beam.draw()); if (shell) { shell.update(delta); if (!shell.active) { shell = null; } } if (shell) shell.draw(); requestAnimationFrame(gameLoop); } function resizeCanvas() { const gameField = document.getElementById('gameField'); canvas.width = gameField.clientWidth; canvas.height = gameField.clientHeight; } shootButton.addEventListener('click', function() { shoot(); }); window.onload = function() { init(); }; window.addEventListener('resize', function() { resizeCanvas(); }); // Включаем выстрел по пробелу window.addEventListener('keydown', function(e) { if (e.code === 'Space') { shoot(); } });
